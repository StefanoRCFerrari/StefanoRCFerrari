#!/bin/bash

HOSTS_FILE="../config/hosts.txt"

if [ ! -f "$HOSTS_FILE" ]; then
    echo "Erro: arquivo config/hosts.txt não encontrado."
    exit 1
fi

DATA=$(date +"%Y-%m-%d")
HORA=$(date +"%H:%M:%S")
RELATORIO="../relatorios/relatorio_${DATA}.txt"

total=0
up=0
down=0
instavel=0

printf "RELATORIO DE CONECTIVIDADE\n" > "$RELATORIO"
printf "Data: %s | Hora: %s\n\n" "$DATA" "$HORA" >> "$RELATORIO"
printf "%-15s %-10s %-10s\n" "HOST" "STATUS" "TENTATIVAS" >> "$RELATORIO"

while read -r host
do
    total=$((total+1))
    tentativa=1
    sucesso=0

    until [ $tentativa -gt 3 ]
    do
        if ping -c 1 -W 1 "$host" > /dev/null 2>&1
        then
            sucesso=1
            break
        fi
        tentativa=$((tentativa+1))
    done

    if [ $sucesso -eq 1 ]; then
        if [ $tentativa -eq 1 ]; then
            status="UP"
            up=$((up+1))
        else
            status="INSTAVEL"
            instavel=$((instavel+1))
        fi
    else
        status="DOWN"
        down=$((down+1))
        tentativa=3
    fi

    printf "%-15s %-10s %d/3\n" "$host" "$status" "$tentativa" >> "$RELATORIO"

done < "$HOSTS_FILE"

printf "\nTOTAL: %d | UP: %d | DOWN: %d | INSTAVEL: %d\n" \
"$total" "$up" "$down" "$instavel" >> "$RELATORIO"

cat "$RELATORIO"
